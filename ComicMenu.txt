        double totalCost = 0;
        if(ironMan.isSelected()){

        }

            totalCost += 1.50;
        }
        if(hulk.isSelected()){
            totalCost += 1.50;
        }
        if(boBo.isSelected()){
            totalCost += 1.50;
        }
        if(mario.isSelected()){

        if(onePiece.isSelected()){

        if(shonenJump.isSelected()){
